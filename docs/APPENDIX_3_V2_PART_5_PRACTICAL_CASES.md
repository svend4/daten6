### B.1. categorize.py

```python
#!/usr/bin/env python3
"""
–ö–∞—Ç–µ–≥–æ—Ä–∏–∑–∞—Ü–∏—è –Ω–æ–≤—ã—Ö —Ñ–∞–π–ª–æ–≤ (Part 1)
"""

import os
import sys
from pathlib import Path

def analyze_file(filepath):
    """–ê–Ω–∞–ª–∏–∑ —Ñ–∞–π–ª–∞ –¥–ª—è –∫–∞—Ç–µ–≥–æ—Ä–∏–∑–∞—Ü–∏–∏"""
    with open(filepath, 'r', encoding='utf-8') as f:
        lines = f.readlines()

    return {
        'filename': os.path.basename(filepath),
        'size_lines': len(lines),
        'size_bytes': os.path.getsize(filepath),
        'estimated_importance': estimate_importance(lines),
        'suggested_integration': suggest_integration_type(len(lines))
    }

def estimate_importance(lines):
    """–ü—Ä–æ—Å—Ç–∞—è —ç–≤—Ä–∏—Å—Ç–∏–∫–∞ –≤–∞–∂–Ω–æ—Å—Ç–∏"""
    # –ú–æ–∂–Ω–æ —É–ª—É—á—à–∏—Ç—å ML –º–æ–¥–µ–ª—å—é
    keywords_critical = ['BVerfG', 'BSG', 'Grundrecht', 'constitutional']
    keywords_high = ['LSG', 'SG', 'important', 'critical']

    score = 50  # baseline

    text = ' '.join(lines).lower()
    score += sum(10 for kw in keywords_critical if kw.lower() in text)
    score += sum(5 for kw in keywords_high if kw.lower() in text)

    return min(score, 100)

def suggest_integration_type(size):
    """–ü—Ä–µ–¥–ª–æ–∂–∏—Ç—å —Ç–∏–ø –∏–Ω—Ç–µ–≥—Ä–∞—Ü–∏–∏"""
    if size < 500:
        return "INSIDE (–º–æ–∂–Ω–æ –≤—Å—Ç–∞–≤–∏—Ç—å –≤ —Å—É—â–µ—Å—Ç–≤—É—é—â–∏–π —Ñ–∞–π–ª)"
    elif size < 2000:
        return "OUTSIDE (—Å–æ–∑–¥–∞—Ç—å –Ω–æ–≤—ã–π —Ñ–∞–π–ª)"
    else:
        return "OUTSIDE + CHUNKING (—Ç—Ä–µ–±—É–µ—Ç—Å—è —Ä–∞–∑–±–∏–µ–Ω–∏–µ)"

if __name__ == '__main__':
    if len(sys.argv) < 2:
        print("Usage: python categorize.py <file1> <file2> ...")
        sys.exit(1)

    for filepath in sys.argv[1:]:
        result = analyze_file(filepath)
        print(f"\nüìÑ {result['filename']}")
        print(f"   –°—Ç—Ä–æ–∫: {result['size_lines']}")
        print(f"   –†–∞–∑–º–µ—Ä: {result['size_bytes']} bytes")
        print(f"   –í–∞–∂–Ω–æ—Å—Ç—å: {result['estimated_importance']}/100")
        print(f"   –†–µ–∫–æ–º–µ–Ω–¥–∞—Ü–∏—è: {result['suggested_integration']}")
```
